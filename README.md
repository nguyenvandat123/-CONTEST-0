**Bài 2. Print number**
```sh
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main() {
    int a;
    long long b;
    char x;
    float c;
    double d;
    
    scanf("%d\n",&a);
    scanf("%lld\n",&b);
    scanf("%c\n",&x);
    scanf("%f\n",&c);
    scanf("%lf\n",&d);
    printf("%d\n",a);
    printf("%lld\n",b);
    printf("%c\n",x);
    printf("%.2f\n",c);
    printf("%.9lf\n",d);

    /* Enter your code here. Read input from STDIN. Print output to STDOUT */    
    return 0;
}
```
